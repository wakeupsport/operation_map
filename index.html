<script>
const sheetURL = "PASTE_LINK_CSV_HERE";

let rawData = [];

fetch(sheetURL)
  .then(res => res.text())
  .then(csv => {
    const rows = csv.split("\n").slice(1);
    rawData = rows.map(r => {
      const [province, type, name, address, phone, note] = r.split(",");
      return { province, type, name, address, phone, note };
    });

    renderProvinces();
  });

function renderProvinces() {
  const provinces = [...new Set(rawData.map(d => d.province))];
  const container = document.getElementById("provinceList");

  provinces.forEach(p => {
    const div = document.createElement("div");
    div.className = "province";
    div.innerText = p;
    div.onclick = () => showProvince(p);
    container.appendChild(div);
  });
}

function showProvince(province) {
  document.getElementById("selectedProvince").innerText = province;

  const filtered = rawData.filter(d => d.province === province);
  const container = document.getElementById("data");
  container.innerHTML = "";

  const types = {
    hotel: "Khách sạn",
    food: "Ăn uống",
    post: "Bưu cục",
    car_rental: "Thuê xe"
  };

  Object.keys(types).forEach(type => {
    const section = filtered.filter(d => d.type === type);
    if (section.length === 0) return;

    const title = document.createElement("h3");
    title.innerText = types[type];
    container.appendChild(title);

    section.forEach(item => {
      const div = document.createElement("div");
      div.className = "card";
      div.innerHTML = `
        <b>${item.name}</b><br>
        ${item.address}<br>
        ${item.phone || ""}<br>
        <i>${item.note || ""}</i>
      `;
      container.appendChild(div);
    });
  });
}
</script>

</body>
</html>

<script>
const sheetURL = "PASTE_CSV_LINK";
let rawData = [];
let currentProvince = "";
let currentFilter = "all";

fetch(sheetURL)
  .then(res => res.text())
  .then(csv => {
    const rows = csv.split("\n").slice(1);
    rawData = rows.map(r => {
      const [province_id, province_name, type, name, address, phone, map_link, note] = r.split(",");
      return { province_id, province_name, type, name, address, phone, map_link, note };
    });

    initMap();
  });

function initMap() {
  document.querySelectorAll("path").forEach(p => {
    p.addEventListener("click", () => {
      document.querySelectorAll("path").forEach(x => x.classList.remove("active"));
      p.classList.add("active");

      currentProvince = p.id;
      render();
    });
  });
}

function setFilter(type) {
  currentFilter = type;
  render();
}

function render() {
  let data = rawData.filter(d => d.province_id === currentProvince);

  if (currentFilter !== "all") {
    data = data.filter(d => d.type === currentFilter);
  }

  if (data.length === 0) return;

  document.getElementById("provinceName").innerText = data[0].province_name;

  const container = document.getElementById("data");
  container.innerHTML = "";

  data.forEach(item => {
    const div = document.createElement("div");
    div.className = "bg-white p-3 mb-2 rounded shadow";

    div.innerHTML = `
      <b>${item.name}</b><br>
      ${item.address}<br>
      ${item.phone || ""}<br>
      <a href="${item.map_link}" target="_blank" class="text-blue-500">Map</a><br>
      <i>${item.note || ""}</i>
    `;

    container.appendChild(div);
  });
}
</script>

</body>

<script>
// LINK CSV của bạn
const sheetURL = "https://docs.google.com/spreadsheets/d/e/2PACX-1vRTHAfXwcrhnhMDq0pgLjdrnIZf8f8xn01kPoKNxDoMTqph4coCQPIe67vcixVmPIjAdE6AV73jMXfh/pub?output=csv";

let rawData = [];
let currentProvince = "";
let currentFilter = "all";

fetch(sheetURL)
  .then(res => res.text())
  .then(csv => {
    const rows = csv.split("\n").slice(1);
    rawData = rows.map(r => {
      const cols = r.split(/,(?=(?:(?:[^"]*"){2})*[^"]*$)/);
      return {
        province_id: cols[0],
        province_name: cols[1],
        type: cols[2],
        name: cols[3],
        address: cols[4],
        phone: cols[5],
        map_link: cols[6],
        note: cols[7]
      };
    });
    initMap();
  });

function initMap(){
  document.querySelectorAll("path").forEach(p=>{
    p.addEventListener("click", ()=>{
      document.querySelectorAll("path").forEach(x=>x.classList.remove("active"));
      p.classList.add("active");
      currentProvince = p.id;
      render();
    });
  });
}

function setFilter(type){
  currentFilter = type;
  render();
}

function render(){
  let data = rawData.filter(d=>d.province_id===currentProvince);
  if(currentFilter !== "all"){
    data = data.filter(d=>d.type===currentFilter);
  }
  if(data.length===0){
    document.getElementById("provinceName").innerText = "Không có dữ liệu";
    document.getElementById("data").innerHTML = "";
    return;
  }
  document.getElementById("provinceName").innerText = data[0].province_name;
  const container = document.getElementById("data");
  container.innerHTML = "";
  data.forEach(item=>{
    const div = document.createElement("div");
    div.className = "bg-white p-3 mb-2 rounded shadow";
    div.innerHTML = `
      <b>${item.name}</b><br>
      ${item.address}<br>
      ${item.phone || ""}<br>
      <a href="${item.map_link}" target="_blank" class="text-blue-500">📍 Map</a><br>
      <i>${item.note || ""}</i>
    `;
    container.appendChild(div);
  });
}

<script>
// Link CSV Google Sheet của bạn
const sheetURL = "https://docs.google.com/spreadsheets/d/e/2PACX-1vRTHAfXwcrhnhMDq0pgLjdrnIZf8f8xn01kPoKNxDoMTqph4coCQPIe67vcixVmPIjAdE6AV73jMXfh/pub?output=csv";

let rawData = [];
let currentProvince = "";
let currentFilter = "all";

fetch(sheetURL)
.then(res => res.text())
.then(csv => {
  const rows = csv.split("\n").slice(1);
  rawData = rows.map(r => {
    const cols = r.split(/,(?=(?:(?:[^"]*"){2})*[^"]*$)/);
    return {
      province_id: cols[0],
      province_name: cols[1],
      type: cols[2],
      name: cols[3],
      address: cols[4],
      phone: cols[5],
      map_link: cols[6],
      note: cols[7]
    };
  });
  initMapClicks();
});

function initMapClicks() {
  document.querySelectorAll("#vnMap path").forEach(p => {
    p.addEventListener("click", () => {
      document.querySelectorAll("#vnMap path").forEach(x => x.classList.remove("active"));
      p.classList.add("active");
      currentProvince = p.id;
      renderData();
    });
  });
}

function setFilter(type) {
  currentFilter = type;
  renderData();
}

function renderData() {
  let data = rawData.filter(d => d.province_id === currentProvince);

  if(currentFilter !== "all") {
    data = data.filter(d => d.type === currentFilter);
  }

  const nameEl = document.getElementById("provinceName");
  const container = document.getElementById("data");
  container.innerHTML = "";

  if(data.length === 0) {
    nameEl.innerText = "Không có dữ liệu";
    return;
  }

  nameEl.innerText = data[0].province_name;

  data.forEach(item => {
    const div = document.createElement("div");
    div.className = "bg-white p-3 mb-2 rounded shadow";
    div.innerHTML = `
      <b>${item.name}</b><br>
      ${item.address}<br>
      ${item.phone || ""}<br>
      <a href="${item.map_link}" target="_blank" class="text-blue-500">📍 Map</a><br>
      <i>${item.note || ""}</i>
    `;
    container.appendChild(div);
  });
}
</script>

// GOOGLE SHEET CSV
const sheetURL = "https://docs.google.com/spreadsheets/d/e/2PACX-1vRTHAfXwcrhnhMDq0pgLjdrnIZf8f8xn01kPoKNxDoMTqph4coCQPIe67vcixVmPIjAdE6AV73jMXfh/pub?output=csv";

let rawData = [];

let currentFilter = "all";

fetch(sheetURL)
  .then(res => res.text())
  .then(csv => {
    const rows = csv.split("\n").slice(1);
    rawData = rows.map(r => {
      const cols = r.split(/,(?=(?:(?:[^"]*"){2})*[^"]*$)/);
      return {
        province_id: cols[0],
        province_name: cols[1],
        type: cols[2],
        name: cols[3],

        phone: cols[5],

        note: cols[7]

function initMapClicks() {
  document.querySelectorAll("#vnMap path").forEach(p => {
    p.addEventListener("click", () => {
      document.querySelectorAll("#vnMap path").forEach(x => x.classList.remove("active"));
      p.classList.add("active");
      currentProvince = p.id;
      renderData();
    });
  });
}

// Filter
function setFilter(type) {
  currentFilter = type;
  renderData();
}

// Render data panel
function renderData() {
  let data = rawData.filter(d => d.province_id === currentProvince);
  if (currentFilter !== "all") data = data.filter(d => d.type === currentFilter);

  const nameEl = document.getElementById("provinceName");
  const container = document.getElementById("data");
  container.innerHTML = "";

  if (!data.length) {
    nameEl.innerText = "Không có dữ liệu";
    return;
  }

  nameEl.innerText = data[0].province_name;
  data.forEach(item => {
    const div = document.createElement("div");
    div.className = "bg-white p-3 mb-2 rounded shadow";
    div.innerHTML = `
      <b>${item.name}</b><br>
      ${item.address}<br>
      ${item.phone || ""}<br>
      <a href="${item.map_link}" target="_blank" class="text-blue-500">📍 Map</a><br>
      <i>${item.note || ""}</i>
    `;
    container.appendChild(div);
  });
}
